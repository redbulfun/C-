\begin{tikzpicture}[node distance=2cm, overlay]
    \begin{scope}[xshift=5cm, yshift=3cm]

    % 定义流程图节点样式
    \tikzstyle{startstop} = [rectangle, rounded corners, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=red!30]
    \tikzstyle{process} = [rectangle, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=orange!30]
    \tikzstyle{decision} = [diamond, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=green!30]
    \tikzstyle{arrow} = [thick,->,>=stealth]

    % 绘制流程图节点
    \node [startstop] (start) {input a kind of data};
    \node [decision, below of=start, yshift=-3cm] (dec1) {whether empty};
    \node [decision, right of=dec1, xshift=3cm] (error1) {Please input again, q to quit};
    \node [startstop, below of=error1, yshift=-2cm] (stop1) {print error};

    \node [process, below right=2cm and 1.5cm of dec2] (pro3) {r\_class};

    \node [startstop, below of=dec2, yshift=-5cm] (stop2) {stop output the point of the list};

    % 绘制流程图箭头连接
    \draw [arrow] (start) -- (dec1);
    \draw [arrow] (dec1) -- node[anchor=south] {Yes} (error1);
    \draw [arrow] (error1) --node[anchor=east] {quit} (stop1);
    \draw [arrow] (error1) |- (start);
    \draw [arrow] (dec1) -- node[anchor=east] {No} (dec2);
    \draw [arrow] (dec2) -| (pro1);
    \draw [arrow] (dec2) -| (pro2);
    \draw [arrow] (dec2) -| (pro3);
